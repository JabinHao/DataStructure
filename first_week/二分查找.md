Position BinarySearch( List L, ElementType X ){
    int first = 1, last = L->Last, center;
    if (L->Data[first]==X)
        return 1;
    else if (L->Data[last]==X)
        return last;
    while (1){
        if (last-first==1)
        return NotFound;
        center = (first+last)/2;
        if (L->Data[center]<X){
            first = center;
        }
        else if (L->Data[center]>X){
            last = center;
        }
        else if (L->Data[center]==X)
            return center;
    }
}
